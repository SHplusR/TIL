class Solution {
    static int answer;
    public int solution(int[] numbers, int target) {
        answer = 0;
        dfs(numbers,0,0,target);
        return answer;
    }
    public void dfs(int[] numbers, int depth, int temp, int target){
        if(depth == numbers.length){
            if(temp == target){
                 answer++;
            }
           return;
        }
        dfs(numbers, depth+1, temp+numbers[depth], target);
        dfs(numbers, depth+1, temp-numbers[depth], target);
            }
    }
